25def write_table( 

26 image_group: zarr.hierarchy.Group, 

27 table_name: str, 

28 table: ad.AnnData, 

29 overwrite: bool = False, 

30 table_type: Optional[str] = None, 

31 table_attrs: Optional[dict[str, Any]] = None, 

32) -> zarr.group: 

33 """ 

34 Write a table to a Zarr group. 

35 

36 This is the general interface that should allow for a smooth coexistence of 

37 tables with different `fractal_table_version` values. Currently only V1 is 

38 defined and implemented. The assumption is that V2 should only change: 

39 

40 1. The lower-level writing function (that is, `_write_table_v2`). 

41 2. The type of the table (which would also reflect into a more general type 

42 hint for `table`, in the current funciton); 

43 3. A different definition of what values of `table_attrs` are valid or 

44 invalid, to be implemented in `_write_table_v2`. 

45 4. Possibly, additional parameters for `_write_table_v2`, which will be 

46 optional parameters of `write_table` (so that `write_table` remains 

47 valid for both V1 and V2). 

48 

49 Args: 

50 image_group: 

51 The image Zarr group where the table will be written. 

52 table_name: 

53 The name of the table. 

54 table: 

55 The table object (currently an AnnData object, for V1). 

56 overwrite: 

57 If `False`, check that the new table does not exist (either as a 

58 zarr sub-group or as part of the zarr-group attributes). In all 

59 cases, propagate parameter to low-level functions, to determine the 

60 behavior in case of an existing sub-group named as in `table_name`. 

61 table_type: `type` attribute for the table; in case `type` is also 

62 present in `table_attrs`, this function argument takes priority. 

63 table_attrs: 

64 If set, overwrite table_group attributes with table_attrs key/value 

65 pairs. If `table_type` is not provided, then `table_attrs` must 

66 include the `type` key. 

67 

68 Returns: 

69 Zarr group of the table. 

70 """ 

71 # Choose which version to use, giving priority to a value that is present 

72 # in table_attrs 

73 version = __FRACTAL_TABLE_VERSION__ 

74 if table_attrs is not None: 

75 try: 

76 version = table_attrs["fractal_table_version"] 

77 except KeyError: 

78 pass 

79 

80 if version == "1": 

81 return _write_table_v1( 

82 image_group, 

83 table_name, 

84 table, 

85 overwrite, 

86 table_type, 

87 table_attrs, 

88 ) 

89 else: 

90 raise NotImplementedError( 

91 f"fractal_table_version='{version}' is not supported" 

92 )
