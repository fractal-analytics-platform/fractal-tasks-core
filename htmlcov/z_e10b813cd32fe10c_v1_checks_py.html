28def check_valid_ROI_indices( 

29 list_indices: list[list[int]], 

30 ROI_table_name: str, 

31) -> None: 

32 """ 

33 Check that list of indices has zero origin on each axis. 

34 

35 See fractal-tasks-core issues #530 and #554. 

36 

37 This helper function is meant to provide informative error messages when 

38 ROI tables created with fractal-tasks-core up to v0.11 are used in v0.12. 

39 This function will be deprecated and removed as soon as the v0.11/v0.12 

40 transition advances. 

41 

42 Note that only `FOV_ROI_table` and `well_ROI_table` have to fulfill this 

43 constraint, while ROI tables obtained through segmentation may have 

44 arbitrary (non-negative) indices. 

45 

46 Args: 

47 list_indices: 

48 Output of `convert_ROI_table_to_indices`; each item is like 

49 `[start_z, end_z, start_y, end_y, start_x, end_x]`. 

50 ROI_table_name: Name of the ROI table. 

51 

52 Raises: 

53 ValueError: 

54 If the table name is `FOV_ROI_table` or `well_ROI_table` and the 

55 minimum value of `start_x`, `start_y` and `start_z` are not all 

56 zero. 

57 """ 

58 if ROI_table_name not in ["FOV_ROI_table", "well_ROI_table"]: 

59 # This validation function only applies to the FOV/well ROI tables 

60 # generated with fractal-tasks-core 

61 return 

62 

63 # Find minimum index along ZYX 

64 min_start_z = min(item[0] for item in list_indices) 

65 min_start_y = min(item[2] for item in list_indices) 

66 min_start_x = min(item[4] for item in list_indices) 

67 

68 # Check that minimum indices are all zero 

69 for ind, min_index in enumerate((min_start_z, min_start_y, min_start_x)): 

70 if min_index != 0: 

71 axis = ["Z", "Y", "X"][ind] 

72 raise ValueError( 

73 f"{axis} component of ROI indices for table `{ROI_table_name}`" 

74 f" do not start with 0, but with {min_index}.\n" 

75 "Hint: As of fractal-tasks-core v0.12, FOV/well ROI " 

76 "tables with non-zero origins (e.g. the ones created with " 

77 "v0.11) are not supported." 

78 )

81def is_ROI_table_valid(*, table_path: str, use_masks: bool) -> Optional[bool]: 

82 """ 

83 Verify some validity assumptions on a ROI table. 

84 

85 This function reflects our current working assumptions (e.g. the presence 

86 of some specific columns); this may change in future versions. 

87 

88 If `use_masks=True`, we verify that the table is a valid 

89 `masking_roi_table` as of table specifications V1; if this check fails, 

90 `use_masks` should be set to `False` upstream in the parent function. 

91 

92 Args: 

93 table_path: Path of the AnnData ROI table to be checked. 

94 use_masks: If `True`, perform some additional checks related to 

95 masked loading. 

96 

97 Returns: 

98 Always `None` if `use_masks=False`, otherwise return whether the table 

99 is valid for masked loading. 

100 """ 

101 

102 table = ad.read_zarr(table_path) 

103 are_ROI_table_columns_valid(table=table) 

104 if not use_masks: 

105 return None 

106 

107 # Check whether the table can be used for masked loading 

108 attrs = zarr.group(table_path).attrs.asdict() 

109 logger.info(f"ROI table at {table_path} has attrs: {attrs}") 

110 try: 

111 MaskingROITableAttrs(**attrs) 

112 logging.info("ROI table can be used for masked loading") 

113 return True 

114 except ValidationError: 

115 logging.info("ROI table cannot be used for masked loading") 

116 return False

119def are_ROI_table_columns_valid(*, table: ad.AnnData) -> None: 

120 """ 

121 Verify some validity assumptions on a ROI table. 

122 

123 This function reflects our current working assumptions (e.g. the presence 

124 of some specific columns); this may change in future versions. 

125 

126 Args: 

127 table: AnnData table to be checked 

128 """ 

129 

130 # Hard constraint: table columns must include some expected ones 

131 columns = [ 

132 "x_micrometer", 

133 "y_micrometer", 

134 "z_micrometer", 

135 "len_x_micrometer", 

136 "len_y_micrometer", 

137 "len_z_micrometer", 

138 ] 

139 for column in columns: 

140 if column not in table.var_names: 

141 raise ValueError(f"Column {column} is not present in ROI table")
