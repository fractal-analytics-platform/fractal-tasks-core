105def check_well_channel_labels(*, well_zarr_path: str) -> None: 

106 """ 

107 Check that the channel labels for a well are unique. 

108 

109 First identify the channel-labels list for each image in the well, then 

110 compare lists and verify their intersection is empty. 

111 

112 Args: 

113 well_zarr_path: path to an OME-NGFF well zarr group. 

114 """ 

115 

116 # Iterate over all images (multiplexing cycles, multi-FOVs, ...) 

117 group = zarr.open_group(well_zarr_path, mode="r+") 

118 image_paths = [image["path"] for image in group.attrs["well"]["images"]] 

119 list_of_channel_lists = [] 

120 for image_path in image_paths: 

121 channels = get_omero_channel_list( 

122 image_zarr_path=f"{well_zarr_path}/{image_path}" 

123 ) 

124 list_of_channel_lists.append(channels[:]) 

125 

126 # For each pair of channel-labels lists, verify they do not overlap 

127 for ind_1, channels_1 in enumerate(list_of_channel_lists): 

128 labels_1 = set([c.label for c in channels_1]) 

129 for ind_2 in range(ind_1): 

130 channels_2 = list_of_channel_lists[ind_2] 

131 labels_2 = set([c.label for c in channels_2]) 

132 intersection = labels_1 & labels_2 

133 if intersection: 

134 hint = ( 

135 "Are you parsing fields of view into separate OME-Zarr " 

136 "images? This could lead to non-unique channel labels, " 

137 "and then could be the reason of the error" 

138 ) 

139 raise ValueError( 

140 "Non-unique channel labels\n" 

141 f"{labels_1=}\n{labels_2=}\n{hint}" 

142 )

145def get_channel_from_image_zarr( 

146 *, 

147 image_zarr_path: str, 

148 label: Optional[str] = None, 

149 wavelength_id: Optional[str] = None, 

150) -> OmeroChannel: 

151 """ 

152 Extract a channel from OME-NGFF zarr attributes. 

153 

154 This is a helper function that combines `get_omero_channel_list` with 

155 `get_channel_from_list`. 

156 

157 Args: 

158 image_zarr_path: Path to an OME-NGFF image zarr group. 

159 label: `label` attribute of the channel to be extracted. 

160 wavelength_id: `wavelength_id` attribute of the channel to be 

161 extracted. 

162 

163 Returns: 

164 A single channel dictionary. 

165 """ 

166 omero_channels = get_omero_channel_list(image_zarr_path=image_zarr_path) 

167 channel = get_channel_from_list( 

168 channels=omero_channels, label=label, wavelength_id=wavelength_id 

169 ) 

170 return channel

173def get_omero_channel_list(*, image_zarr_path: str) -> list[OmeroChannel]: 

174 """ 

175 Extract the list of channels from OME-NGFF zarr attributes. 

176 

177 Args: 

178 image_zarr_path: Path to an OME-NGFF image zarr group. 

179 

180 Returns: 

181 A list of channel dictionaries. 

182 """ 

183 group = zarr.open_group(image_zarr_path, mode="r+") 

184 channels_dicts = group.attrs["omero"]["channels"] 

185 channels = [OmeroChannel(**c) for c in channels_dicts] 

186 return channels

189def get_channel_from_list( 

190 *, 

191 channels: list[OmeroChannel], 

192 label: Optional[str] = None, 

193 wavelength_id: Optional[str] = None, 

194) -> OmeroChannel: 

195 """ 

196 Find matching channel in a list. 

197 

198 Find the channel that has the required values of `label` and/or 

199 `wavelength_id`, and identify its positional index (which also 

200 corresponds to its index in the zarr array). 

201 

202 Args: 

203 channels: A list of channel dictionary, where each channel includes (at 

204 least) the `label` and `wavelength_id` keys. 

205 label: The label to look for in the list of channels. 

206 wavelength_id: The wavelength_id to look for in the list of channels. 

207 

208 Returns: 

209 A single channel dictionary. 

210 """ 

211 

212 # Identify matching channels 

213 if label: 

214 if wavelength_id: 

215 # Both label and wavelength_id are specified 

216 matching_channels = [ 

217 c 

218 for c in channels 

219 if (c.label == label and c.wavelength_id == wavelength_id) 

220 ] 

221 else: 

222 # Only label is specified 

223 matching_channels = [c for c in channels if c.label == label] 

224 else: 

225 if wavelength_id: 

226 # Only wavelength_id is specified 

227 matching_channels = [ 

228 c for c in channels if c.wavelength_id == wavelength_id 

229 ] 

230 else: 

231 # Neither label or wavelength_id are specified 

232 raise ValueError( 

233 "get_channel requires at least one in {label,wavelength_id} " 

234 "arguments" 

235 ) 

236 

237 # Verify that there is one and only one matching channel 

238 if len(matching_channels) == 0: 

239 required_match = [f"{label=}", f"{wavelength_id=}"] 

240 required_match_string = " and ".join( 

241 [x for x in required_match if "None" not in x] 

242 ) 

243 raise ChannelNotFoundError( 

244 f"ChannelNotFoundError: No channel found in {channels}" 

245 f" for {required_match_string}" 

246 ) 

247 if len(matching_channels) > 1: 

248 raise ValueError(f"Inconsistent set of channels: {channels}") 

249 

250 channel = matching_channels[0] 

251 channel.index = channels.index(channel) 

252 return channel

255def define_omero_channels( 

256 *, 

257 channels: list[OmeroChannel], 

258 bit_depth: int, 

259 label_prefix: Optional[str] = None, 

260) -> list[dict[str, Union[str, int, bool, dict[str, int]]]]: 

261 """ 

262 Update a channel list to use it in the OMERO/channels metadata. 

263 

264 Given a list of channel dictionaries, update each one of them by: 

265 1. Adding a label (if missing); 

266 2. Adding a set of OMERO-specific attributes; 

267 3. Discarding all other attributes. 

268 

269 The `new_channels` output can be used in the `attrs["omero"]["channels"]` 

270 attribute of an image group. 

271 

272 Args: 

273 channels: A list of channel dictionaries (each one must include the 

274 `wavelength_id` key). 

275 bit_depth: bit depth. 

276 label_prefix: TBD 

277 

278 Returns: 

279 `new_channels`, a new list of consistent channel dictionaries that 

280 can be written to OMERO metadata. 

281 """ 

282 

283 new_channels = [c.copy(deep=True) for c in channels] 

284 default_colors = ["00FFFF", "FF00FF", "FFFF00"] 

285 

286 for channel in new_channels: 

287 wavelength_id = channel.wavelength_id 

288 

289 # If channel.label is None, set it to a default value 

290 if channel.label is None: 

291 default_label = wavelength_id 

292 if label_prefix: 

293 default_label = f"{label_prefix}_{default_label}" 

294 logging.warning( 

295 f"Missing label for {channel=}, using {default_label=}" 

296 ) 

297 channel.label = default_label 

298 

299 # If channel.color is None, set it to a default value (use the default 

300 # ones for the first three channels, or gray otherwise) 

301 if channel.color is None: 

302 try: 

303 channel.color = default_colors.pop() 

304 except IndexError: 

305 channel.color = "808080" 

306 

307 # Set channel.window attribute 

308 if channel.window: 

309 channel.window.min = 0 

310 channel.window.max = 2**bit_depth - 1 

311 

312 # Check that channel labels are unique for this image 

313 labels = [c.label for c in new_channels] 

314 if len(set(labels)) < len(labels): 

315 raise ValueError(f"Non-unique labels in {new_channels=}") 

316 

317 new_channels_dictionaries = [ 

318 c.dict(exclude={"index"}, exclude_unset=True) for c in new_channels 

319 ] 

320 

321 return new_channels_dictionaries
