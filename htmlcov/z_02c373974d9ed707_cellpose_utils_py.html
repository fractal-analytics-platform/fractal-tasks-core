33class CellposeCustomNormalizer(BaseModel): 

34 """ 

35 Validator to handle different normalization scenarios for Cellpose models 

36 

37 If `type="default"`, then Cellpose default normalization is 

38 used and no other parameters can be specified. 

39 If `type="no_normalization"`, then no normalization is used and no 

40 other parameters can be specified. 

41 If `type="custom"`, then either percentiles or explicit integer 

42 bounds can be applied. 

43 

44 Attributes: 

45 type: 

46 One of `default` (Cellpose default normalization), `custom` 

47 (using the other custom parameters) or `no_normalization`. 

48 lower_percentile: Specify a custom lower-bound percentile for rescaling 

49 as a float value between 0 and 100. Set to 1 to run the same as 

50 default). You can only specify percentiles or bounds, not both. 

51 upper_percentile: Specify a custom upper-bound percentile for rescaling 

52 as a float value between 0 and 100. Set to 99 to run the same as 

53 default, set to e.g. 99.99 if the default rescaling was too harsh. 

54 You can only specify percentiles or bounds, not both. 

55 lower_bound: Explicit lower bound value to rescale the image at. 

56 Needs to be an integer, e.g. 100. 

57 You can only specify percentiles or bounds, not both. 

58 upper_bound: Explicit upper bound value to rescale the image at. 

59 Needs to be an integer, e.g. 2000. 

60 You can only specify percentiles or bounds, not both. 

61 """ 

62 

63 type: Literal["default", "custom", "no_normalization"] = "default" 

64 lower_percentile: Optional[float] = Field(None, ge=0, le=100) 

65 upper_percentile: Optional[float] = Field(None, ge=0, le=100) 

66 lower_bound: Optional[int] = None 

67 upper_bound: Optional[int] = None 

68 

69 # In the future, add an option to allow using precomputed percentiles 

70 # that are stored in OME-Zarr histograms and use this pydantic model that 

71 # those histograms actually exist 

72 

73 @model_validator(mode="after") 

74 def validate_conditions(self: Self) -> Self: 

75 # Extract values 

76 type = self.type 

77 lower_percentile = self.lower_percentile 

78 upper_percentile = self.upper_percentile 

79 lower_bound = self.lower_bound 

80 upper_bound = self.upper_bound 

81 

82 # Verify that custom parameters are only provided when type="custom" 

83 if type != "custom": 

84 if lower_percentile is not None: 

85 raise ValueError( 

86 f"Type='{type}' but {lower_percentile=}. " 

87 "Hint: set type='custom'." 

88 ) 

89 if upper_percentile is not None: 

90 raise ValueError( 

91 f"Type='{type}' but {upper_percentile=}. " 

92 "Hint: set type='custom'." 

93 ) 

94 if lower_bound is not None: 

95 raise ValueError( 

96 f"Type='{type}' but {lower_bound=}. " 

97 "Hint: set type='custom'." 

98 ) 

99 if upper_bound is not None: 

100 raise ValueError( 

101 f"Type='{type}' but {upper_bound=}. " 

102 "Hint: set type='custom'." 

103 ) 

104 

105 # The only valid options are: 

106 # 1. Both percentiles are set and both bounds are unset 

107 # 2. Both bounds are set and both percentiles are unset 

108 are_percentiles_set = ( 

109 lower_percentile is not None, 

110 upper_percentile is not None, 

111 ) 

112 are_bounds_set = ( 

113 lower_bound is not None, 

114 upper_bound is not None, 

115 ) 

116 if len(set(are_percentiles_set)) != 1: 

117 raise ValueError( 

118 "Both lower_percentile and upper_percentile must be set " 

119 "together." 

120 ) 

121 if len(set(are_bounds_set)) != 1: 

122 raise ValueError( 

123 "Both lower_bound and upper_bound must be set together" 

124 ) 

125 if lower_percentile is not None and lower_bound is not None: 

126 raise ValueError( 

127 "You cannot set both explicit bounds and percentile bounds " 

128 "at the same time. Hint: use only one of the two options." 

129 ) 

130 

131 return self

144class CellposeModelParams(BaseModel): 

145 """ 

146 Advanced Cellpose Model Parameters 

147 

148 Attributes: 

149 cellprob_threshold: Parameter of `CellposeModel.eval` method. Valid 

150 values between -6 to 6. From Cellpose documentation: "Decrease this 

151 threshold if cellpose is not returning as many ROIs as you'd 

152 expect. Similarly, increase this threshold if cellpose is returning 

153 too ROIs particularly from dim areas." 

154 flow_threshold: Parameter of `CellposeModel.eval` method. Valid 

155 values between 0.0 and 1.0. From Cellpose documentation: "Increase 

156 this threshold if cellpose is not returning as many ROIs as you'd 

157 expect. Similarly, decrease this threshold if cellpose is returning 

158 too many ill-shaped ROIs." 

159 anisotropy: Ratio of the pixel sizes along Z and XY axis (ignored if 

160 the image is not three-dimensional). If unset, it is inferred from 

161 the OME-NGFF metadata. 

162 min_size: Parameter of `CellposeModel` class. Minimum size of the 

163 segmented objects (in pixels). Use `-1` to turn off the size 

164 filter. 

165 augment: Parameter of `CellposeModel` class. Whether to use cellpose 

166 augmentation to tile images with overlap. 

167 net_avg: Parameter of `CellposeModel` class. Whether to use cellpose 

168 net averaging to run the 4 built-in networks (useful for `nuclei`, 

169 `cyto` and `cyto2`, not sure it works for the others). 

170 use_gpu: If `False`, always use the CPU; if `True`, use the GPU if 

171 possible (as defined in `cellpose.core.use_gpu()`) and fall-back 

172 to the CPU otherwise. 

173 batch_size: number of 224x224 patches to run simultaneously on the GPU 

174 (can make smaller or bigger depending on GPU memory usage) 

175 invert: invert image pixel intensity before running network (if True, 

176 image is also normalized) 

177 tile: tiles image to ensure GPU/CPU memory usage limited (recommended) 

178 tile_overlap: fraction of overlap of tiles when computing flows 

179 resample: run dynamics at original image size (will be slower but 

180 create more accurate boundaries) 

181 interp: interpolate during 2D dynamics (not available in 3D) 

182 (in previous versions it was False, now it defaults to True) 

183 stitch_threshold: if stitch_threshold>0.0 and not do_3D and equal 

184 image sizes, masks are stitched in 3D to return volume segmentation 

185 """ 

186 

187 cellprob_threshold: float = 0.0 

188 flow_threshold: float = 0.4 

189 anisotropy: Optional[float] = None 

190 min_size: int = 15 

191 augment: bool = False 

192 net_avg: bool = False 

193 use_gpu: bool = True 

194 batch_size: int = 8 

195 invert: bool = False 

196 tile: bool = True 

197 tile_overlap: float = 0.1 

198 resample: bool = True 

199 interp: bool = True 

200 stitch_threshold: float = 0.0

294def _normalize_cellpose_channels( 

295 x: np.ndarray, 

296 channels: list[int], 

297 normalize: CellposeCustomNormalizer, 

298 normalize2: CellposeCustomNormalizer, 

299) -> np.ndarray: 

300 """ 

301 Normalize a cellpose input array by channel. 

302 

303 Args: 

304 x: 4D numpy array. 

305 channels: Which channels to use. If only one channel is provided, `[0, 

306 0]` should be used. If two channels are provided (the first 

307 dimension of `x` has length of 2), `[1, 2]` should be used 

308 (`x[0, :, :,:]` contains the membrane channel and 

309 `x[1, :, :, :]` contains the nuclear channel). 

310 normalize: By default, data is normalized so 0.0=1st percentile and 

311 1.0=99th percentile of image intensities in each channel. 

312 This automatic normalization can lead to issues when the image to 

313 be segmented is very sparse. You can turn off the default 

314 rescaling. With the "custom" option, you can either provide your 

315 own rescaling percentiles or fixed rescaling upper and lower 

316 bound integers. 

317 normalize2: Normalization options for channel 2. If one channel is 

318 normalized with default settings, both channels need to be 

319 normalized with default settings. 

320 

321 """ 

322 # Optionally perform custom normalization 

323 # normalize channels separately, if normalize2 is provided: 

324 if channels == [1, 2]: 

325 # If run in single channel mode, fails (specified as channel = [0, 0]) 

326 if (normalize.type == "default") != (normalize2.type == "default"): 

327 raise ValueError( 

328 "ERROR in normalization:" 

329 f" {normalize.type=} and {normalize2.type=}." 

330 " Either both need to be 'default', or none of them." 

331 ) 

332 if normalize.type == "custom": 

333 x[channels[0] - 1 : channels[0]] = normalized_img( 

334 x[channels[0] - 1 : channels[0]], 

335 lower_p=normalize.lower_percentile, 

336 upper_p=normalize.upper_percentile, 

337 lower_bound=normalize.lower_bound, 

338 upper_bound=normalize.upper_bound, 

339 ) 

340 if normalize2.type == "custom": 

341 x[channels[1] - 1 : channels[1]] = normalized_img( 

342 x[channels[1] - 1 : channels[1]], 

343 lower_p=normalize2.lower_percentile, 

344 upper_p=normalize2.upper_percentile, 

345 lower_bound=normalize2.lower_bound, 

346 upper_bound=normalize2.upper_bound, 

347 ) 

348 

349 # otherwise, use first normalize to normalize all channels: 

350 else: 

351 if normalize.type == "custom": 

352 x = normalized_img( 

353 x, 

354 lower_p=normalize.lower_percentile, 

355 upper_p=normalize.upper_percentile, 

356 lower_bound=normalize.lower_bound, 

357 upper_bound=normalize.upper_bound, 

358 ) 

359 

360 return x

363def normalized_img( 

364 img: np.ndarray, 

365 axis: int = -1, 

366 invert: bool = False, 

367 lower_p: float = 1.0, 

368 upper_p: float = 99.0, 

369 lower_bound: Optional[int] = None, 

370 upper_bound: Optional[int] = None, 

371): 

372 """normalize each channel of the image so that so that 0.0=lower percentile 

373 or lower bound and 1.0=upper percentile or upper bound of image intensities. 

374 

375 The normalization can result in values < 0 or > 1 (no clipping). 

376 

377 Based on https://github.com/MouseLand/cellpose/blob/4f5661983c3787efa443bbbd3f60256f4fd8bf53/cellpose/transforms.py#L375 # noqa E501 

378 

379 optional inversion 

380 

381 Parameters 

382 ------------ 

383 

384 img: ND-array (at least 3 dimensions) 

385 

386 axis: channel axis to loop over for normalization 

387 

388 invert: invert image (useful if cells are dark instead of bright) 

389 

390 lower_p: Lower percentile for rescaling 

391 

392 upper_p: Upper percentile for rescaling 

393 

394 lower_bound: Lower fixed-value used for rescaling 

395 

396 upper_bound: Upper fixed-value used for rescaling 

397 

398 Returns 

399 --------------- 

400 

401 img: ND-array, float32 

402 normalized image of same size 

403 

404 """ 

405 if img.ndim < 3: 405 ↛ 406line 405 didn't jump to line 406 because the condition on line 405 was never true

406 error_message = "Image needs to have at least 3 dimensions" 

407 logger.critical(error_message) 

408 raise ValueError(error_message) 

409 

410 img = img.astype(np.float32) 

411 img = np.moveaxis(img, axis, 0) 

412 for k in range(img.shape[0]): 

413 if lower_p is not None: 

414 # ptp can still give nan's with weird images 

415 i99 = np.percentile(img[k], upper_p) 

416 i1 = np.percentile(img[k], lower_p) 

417 if i99 - i1 > +1e-3: # np.ptp(img[k]) > 1e-3: 417 ↛ 424line 417 didn't jump to line 424 because the condition on line 417 was always true

418 img[k] = normalize_percentile( 

419 img[k], lower=lower_p, upper=upper_p 

420 ) 

421 if invert: 421 ↛ 422line 421 didn't jump to line 422 because the condition on line 421 was never true

422 img[k] = -1 * img[k] + 1 

423 else: 

424 img[k] = 0 

425 elif lower_bound is not None: 425 ↛ 435line 425 didn't jump to line 435 because the condition on line 425 was always true

426 if upper_bound - lower_bound > +1e-3: 426 ↛ 433line 426 didn't jump to line 433 because the condition on line 426 was always true

428 img[k], lower=lower_bound, upper=upper_bound 

429 )

431 img[k] = -1 * img[k] + 1 

432 else: 

433 img[k] = 0 

434 else: 

435 raise ValueError("No normalization method specified") 

436 img = np.moveaxis(img, 0, axis) 

437 return img

440def normalize_percentile(Y: np.ndarray, lower: float = 1, upper: float = 99): 

441 """normalize image so 0.0 is lower percentile and 1.0 is upper percentile 

442 Percentiles are passed as floats (must be between 0 and 100) 

443 

444 Args: 

445 Y: The image to be normalized 

446 lower: Lower percentile 

447 upper: Upper percentile 

448 

449 """ 

450 X = Y.copy() 

451 x01 = np.percentile(X, lower) 

452 x99 = np.percentile(X, upper) 

453 X = (X - x01) / (x99 - x01) 

454 return X

462 lower: Lower normalization value
