48def _write_registered_ngio_image( 

49 source_ome_zarr: OmeZarrContainer, 

50 new_zarr_url: str, 

51 roi_pairs: list[tuple[Roi, Roi]], 

52) -> OmeZarrContainer: 

53 """Write a registered OME-Zarr image to disk using pre-computed ROI pairs. 

54 

55 Creates a new image container derived from the source (same shape, dtype, 

56 and metadata, initialised to zeros), writes image data from each acquisition 

57 ROI into the corresponding reference ROI position, then builds the pyramid 

58 using linear interpolation. 

59 

60 Args: 

61 source_ome_zarr: Source image container (current acquisition). 

62 new_zarr_url: Path where the new registered image will be written. 

63 roi_pairs: List of (acq_roi, ref_roi) tuples. For each pair the data 

64 is read from `acq_roi` in the source and written to `ref_roi` in 

65 the new image. Regions not covered by any pair remain zero. 

66 

67 Returns: 

68 The newly created OmeZarrContainer. 

69 """ 

70 new_ome_zarr = source_ome_zarr.derive_image(new_zarr_url, overwrite=True) 

71 source_image = source_ome_zarr.get_image() 

72 new_image = new_ome_zarr.get_image() 

73 for acq_roi, ref_roi in roi_pairs: 

74 patch = source_image.get_roi(acq_roi) 

75 new_image.set_roi(ref_roi, patch) 

76 new_image.consolidate(order="linear") 

77 return new_ome_zarr
